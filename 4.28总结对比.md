-- 消息表
CREATE TABLE session_messages (
  id TEXT PRIMARY KEY,
  session_id TEXT FOREIGN KEY,
  role TEXT,  -- 'user' / 'assistant' / 'system'
  content TEXT,
  tokens_used INT,
  created_at DATETIME,
  FOREIGN KEY (session_id) REFERENCES media_sessions(id)
);

-- 工作区表
CREATE TABLE workspaces (
  id TEXT PRIMARY KEY,
  name TEXT NOT NULL,
  mode TEXT,  -- 'image' / 'video'
  prompt TEXT,
  params TEXT,  -- JSON: {size, quality, format, ...}
  status TEXT,  -- 'idle' / 'generating' / 'completed' / 'failed'
  created_at DATETIME,
  updated_at DATETIME
);

-- 生成结果表
CREATE TABLE generation_results (
  id TEXT PRIMARY KEY,
  workspace_id TEXT FOREIGN KEY,
  file_path TEXT,
  base64_preview TEXT,
  metadata TEXT,  -- JSON: {width, height, model_id, ...}
  created_at DATETIME,
  FOREIGN KEY (workspace_id) REFERENCES workspaces(id)
);
